const nodes = [];
let head = null;

function addFront(a) {
  nodes.push([a, head]);
  head = nodes.length - 1;
}

function print() {
  let text = '';
  let i = head;
  while (i !== null) {
    console.log(nodes[i][0]);
    text += nodes[i][0] + '\n';
    i = nodes[i][1];
  }
  document.getElementById('out').textContent = text;
}
